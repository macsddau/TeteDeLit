314void loop() {
316 int state;
317
319 for (int sw = 0; sw < NUM_SWITCHES; sw++) {
321 state = digitalRead(switchPins[sw]);
322
324 if (state != switchStates[sw]) {
326 if (state == LOW) {
328 switchTimes[sw] = millis();
329 } else {
331 if (millis() - switchTimes[sw] < LONG_PULSE) {
333 switch(lightStates[sw]) {
334 case LIGHT_OFF:
335 setLightNight(*leds, numLeds, NUM_STRIPS, *nightPatterns);
336 lightStates[sw] = LIGHT_NIGHT;
337 break;
338 default:
339 setLightOff(*leds, numLeds, NUM_STRIPS);
340 lightStates[sw] = LIGHT_OFF;
341 }
342 } else {
344 switch(lightStates[sw]) {
345 case LIGHT_ON:
346 setLightNight(*leds, numLeds, NUM_STRIPS, *nightPatterns);
347 lightStates[sw] = LIGHT_NIGHT;
348 break;
349 default:
350 setLightOn(*leds, numLeds, NUM_STRIPS);
351 lightStates[sw] = LIGHT_ON;
352 }
353 }
354 }
355
357 switchStates[sw] = state;
358 }
359 }
360
362 FastLED.show();
363}
