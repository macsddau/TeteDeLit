241void loop() {
243 int state;
244
246 for (int sw = 0; sw < NUM_SWITCHES; sw++) {
248 state = digitalRead(switchPins[sw]);
249
251 if (state != switchStates[sw]) {
253 if (state == LOW) {
255 switchTimes[sw] = millis();
256 } else {
258 if (millis() - switchTimes[sw] < LONG_PULSE) {
260 switch(lightStates[sw]) {
261 case LIGHT_OFF:
262 setLightNight(*leds, numLeds, NUM_STRIPS, *nightPatterns);
263 lightStates[sw] = LIGHT_NIGHT;
264 break;
265 default:
266 setLightOff(*leds, numLeds, NUM_STRIPS);
267 lightStates[sw] = LIGHT_OFF;
268 }
269 } else {
271 switch(lightStates[sw]) {
272 case LIGHT_ON:
273 setLightNight(*leds, numLeds, NUM_STRIPS, *nightPatterns);
274 lightStates[sw] = LIGHT_NIGHT;
275 break;
276 default:
277 setLightOn(*leds, numLeds, NUM_STRIPS);
278 lightStates[sw] = LIGHT_ON;
279 }
280 }
281 }
282
284 switchStates[sw] = state;
285 }
286 }
287
289 FastLED.show();
